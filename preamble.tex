  \thispagestyle{empty}
  % Draw the custom title block using tikz
  
  \begin{tikzpicture}[remember picture, overlay]
    
    % 1. The blue background banner (spans full width, 2cm tall)
    \fill[CeHDIBlue] (current page.north west) rectangle ++(\paperwidth, -2cm);

    % 2. The title text
    \node[
        anchor=north, % Anchor the node from its top-center
        text=fgcolor,
        font=\LARGE\bfseries\sffamily,
        text width=\paperwidth-6cm, % Width minus padding so it doesn't overlap logos
        align=center % Horizontally center the text
    ] 
    at ($(current page.north) + (0cm, -0.2cm)$) % Position at top-center, dropped down by 0.4cm
    {\@title};

    % 3. The logo in the top left
    \node[
        anchor=north west, 
    ] 
    at ($(current page.north west) + (0cm, -0.45cm)$)

    % 4. The flag in the top right
    \node[
        anchor=north east, 
    ]
    at ($(current page.north east) + (0cm, -0.15cm)$)

  \end{tikzpicture}
  
  % Bring the country name up into the fill block
